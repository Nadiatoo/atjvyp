## 问题1：周期评分权重动态调整

### 当前问题
春播30%、夏长40%是固定权重，但不同市场环境下应该不同。

### 解决方案：动态权重系统

```python
class DynamicWeightSystem:
    """根据市场环境动态调整周期评分权重"""
    
    def adjust_weights(self, market_context):
        """
        基础权重：
        - 春播信号：30%
        - 夏长信号：40%
        - 秋收信号：20%
        - 冬藏信号：10%
        
        根据市场环境动态调整
        """
        base_weights = {
            'spring': 0.30,  # 春播
            'summer': 0.40,  # 夏长
            'autumn': 0.20,  # 秋收
            'winter': 0.10   # 冬藏
        }
        
        # 根据当前市场状态调整
        adjustments = {}
        
        # 场景1：量化主导市场（2024-2025特征）
        if market_context['quant_dominance'] > 0.5:
            # 量化时代，周期缩短，春播/秋收信号更重要
            adjustments = {
                'spring': +0.05,   # 35%
                'summer': -0.10,   # 30%
                'autumn': +0.05,   # 25%
                'winter': 0        # 10%
            }
        
        # 场景2：机构主导市场（2021年特征）
        elif market_context['institution_dominance'] > 0.6:
            # 机构时代，夏长更持久
            adjustments = {
                'spring': -0.05,   # 25%
                'summer': +0.10,   # 50%
                'autumn': -0.05,   # 15%
                'winter': 0        # 10%
            }
        
        # 场景3：游资主导市场（2015年特征）
        elif market_context['retail_dominance'] > 0.5:
            # 游资时代，冬藏/春播更极端
            adjustments = {
                'spring': +0.10,   # 40%
                'summer': -0.05,   # 35%
                'autumn': -0.05,   # 15%
                'winter': 0        # 10%
            }
        
        # 场景4：流动性极度充裕（3万亿+）
        if market_context['liquidity'] > 30000:
            # 钱多的时候，夏长权重增加
            adjustments['summer'] = adjustments.get('summer', 0) + 0.05
            adjustments['spring'] = adjustments.get('spring', 0) - 0.05
        
        # 场景5：流动性枯竭（2万亿以下）
        elif market_context['liquidity'] < 20000:
            # 钱少的时候，冬藏/春播权重增加
            adjustments['winter'] = adjustments.get('winter', 0) + 0.05
            adjustments['summer'] = adjustments.get('summer', 0) - 0.05
        
        # 应用调整
        final_weights = {}
        for key in base_weights:
            final_weights[key] = base_weights[key] + adjustments.get(key, 0)
        
        # 归一化（确保总和=1）
        total = sum(final_weights.values())
        for key in final_weights:
            final_weights[key] /= total
        
        return final_weights
```

| 市场环境 | 春播 | 夏长 | 秋收 | 冬藏 | 适用时期 |
|:---:|:---:|:---:|:---:|:---:|:---|
| **基础权重** | 30% | 40% | 20% | 10% | 通用 |
| **量化主导** | 35% | 30% | 25% | 10% | 2024-2025 |
| **机构主导** | 25% | 50% | 15% | 10% | 2021年 |
| **游资主导** | 40% | 35% | 15% | 10% | 2015年 |
| **流动性充裕** | 25% | 45% | 20% | 10% | 3万亿+ |
| **流动性枯竭** | 35% | 30% | 20% | 15% | 2万亿以下 |

**当前市场（2026年2月）判断**：
- 量化占比：~30%（量化主导）
- 流动性：2.5万亿（中等）

## 问题3：发酵期龙头识别优化

### 当前问题
发酵期多家竞争，不确定性高。

### 解决方案：发酵期龙头筛选漏斗

```python
class DragonHeadScreeningSystem:
    """发酵期龙头筛选系统"""
    
    def screen fermentation_stage(self, theme_stocks, day1_data):
        """
        首板/二板阶段筛选真龙头
        """
        candidates = []
        
        # 第一层：首板质量筛选（淘汰50%）
        for stock in theme_stocks:
            if self.check_first_limit_quality(stock, day1_data):
                candidates.append(stock)
        
        # 第二层：封单强度排序（淘汰30%）
        candidates = sorted(candidates, 
                          key=lambda x: x['封单金额'], 
                          reverse=True)[:len(candidates)*0.7]
        
        # 第三层：带动性测试（淘汰20%）
        candidates = [s for s in candidates 
                     if self.check_leadership(s, theme_stocks)]
        
        # 第四层：题材正宗性验证（淘汰10%）
        candidates = [s for s in candidates 
                     if self.check_theme_authenticity(s) > 0.7]  # 业务占比>70%
        
        # 第五层：盘子适中性（最终筛选）
        candidates = [s for s in candidates 
                     if 30 <= s['流通市值'] <= 200]
        
        return candidates[:3]  # 返回Top3候选
    
    def check_first_limit_quality(self, stock, data):
        """首板质量检查"""
        checks = {
            '开盘涨幅': data['开盘涨幅'] < 3,  # 不能开太高
            '涨停时间': data['涨停时间'] < '10:30',  # 早盘涨停
            '封单比例': data['封单金额'] / data['流通市值'] > 0.005,  # 封单>0.5%
            '换手率': 3 < data['换手率'] < 15,  # 换手适中
            '前期涨幅': data['20日涨幅'] < 30,  # 不是高位补涨
        }
        return sum(checks.values()) >= 4  # 满足4个以上条件
    
    def check_leadership(self, stock, all_stocks):
        """带动性检查：涨停后是否带动跟风"""
        stock_time = stock['涨停时间']
        
        # 检查stock涨停后30分钟内，多少跟风股涨停
        follower_count = 0
        for s in all_stocks:
            if s['涨停时间'] > stock_time and s['涨停时间'] < stock_time + 30:
                follower_count += 1
        
        return follower_count >= 2  # 带动2个以上跟风
```

## 问题4：中军动态调整机制

### 当前问题
中军会换，比如商业航天从航发展到中国卫通。

### 解决方案：中军动态跟踪系统

```python
class ZhongjunTrackingSystem:
    """中军动态跟踪与切换系统"""
    
    def __init__(self):
        self.zhongjun_candidates = []  # 中军候选池
        self.current_zhongjun = None   # 当前中军
        self.switch_threshold = 0.15   # 切换阈值15%
    
    def update_zhongjun_candidates(self, theme_stocks):
        """每日更新中军候选池"""
        candidates = []
        
        for stock in theme_stocks:
            # 中军三维度筛选
            if (stock['流通市值'] > 200 and  # 市值>200亿
                stock['日均成交'] > 10 and    # 成交>10亿
                stock['业务占比'] > 0.5):     # 业务占比>50%
                
                # 计算中军强度分
                strength_score = self.calculate_zhongjun_strength(stock)
                candidates.append({
                    '股票': stock,
                    '强度分': strength_score,
                    '5日趋势': stock['5日涨幅'],
                    '10日趋势': stock['10日涨幅']
                })
        
        # 按强度分排序
        self.zhongjun_candidates = sorted(candidates, 
                                         key=lambda x: x['强度分'], 
                                         reverse=True)
        
        return self.zhongjun_candidates
    
    def calculate_zhongjun_strength(self, stock):
        """计算中军强度分"""
        score = 0
        
        # 趋势稳定性（40分）
        if stock['5日涨幅'] > 0 and stock['10日涨幅'] > 0:
            score += 40
        elif stock['5日涨幅'] > 0 or stock['10日涨幅'] > 0:
            score += 20
        
        # 成交量能（30分）
        if stock['日均成交'] > 20:
            score += 30
        elif stock['日均成交'] > 10:
            score += 20
        
        # 业务正宗性（20分）
        if stock['业务占比'] > 0.7:
            score += 20
        elif stock['业务占比'] > 0.5:
            score += 10
        
        # 机构持仓（10分）
        if stock['机构持仓'] > 0.3:
            score += 10
        
        return score
    
    def check_zhongjun_switch(self):
        """检查是否需要切换中军"""
        if not self.current_zhongjun or not self.zhongjun_candidates:
            return False, None
        
        # 当前中军排名
        current_rank = None
        for i, c in enumerate(self.zhongjun_candidates):
            if c['股票']['代码'] == self.current_zhongjun['代码']:
                current_rank = i
                break
        
        # 如果当前中军掉出前3，考虑切换
        if current_rank is None or current_rank >= 3:
            new_zhongjun = self.zhongjun_candidates[0]
            
            # 检查新中军是否强15%以上
            if new_zhongjun['强度分'] > self.current_zhongjun.get('强度分', 0) * 1.15:
                return True, new_zhongjun
        
        return False, None
```

class SpringBuyPointCalculator:
    """春播期买点精确计算"""
    
    def calculate_first_buy(self, stock_data):
        """
        春播第一买：底部放量突破颈线
        计算买入价格和仓位
        """
        # 颈线位置 = 近期高点连线
        neckline = self.calculate_neckline(stock_data['近期高点'])
        
        # 买点 = 突破颈线 + 3%确认
        buy_price = neckline * 1.03
        
        # 止损位 = 颈线下方5%
        stop_loss = neckline * 0.95
        
        # 仓位 = 首仓10%
        position = 0.10
        
        # 风险收益比检查
        risk = buy_price - stop_loss
        reward = self.calculate_target_price(stock_data) - buy_price
        
        if reward / risk < 2:  # 风险收益比<2，放弃
            return None
        
        return {
            'type': '春播第一买',
            'buy_price': round(buy_price, 2),
            'stop_loss': round(stop_loss, 2),
            'target_price': round(self.calculate_target_price(stock_data), 2),
            'position': position,
            'risk_reward_ratio': round(reward / risk, 2)
        }
    
    def calculate_second_buy(self, stock_data):
        """
        春播第二买：回踩颈线企稳
        """
        neckline = self.calculate_neckline(stock_data['近期高点'])
        
        # 回踩位置 = 颈线附近（-2%到+2%）
        if abs(stock_data['当前价格'] - neckline) / neckline < 0.02:
            # 企稳信号：缩量+小阳线
            if stock_data['成交量'] < stock_data['5日均量'] * 0.7 and \
               stock_data['涨跌幅'] > 0:
                
                buy_price = stock_data['当前价格']
                stop_loss = neckline * 0.95
                position = 0.20  # 加仓至30%
                
                return {
                    'type': '春播第二买',
                    'buy_price': round(buy_price, 2),
                    'stop_loss': round(stop_loss, 2),
                    'position': position
                }
        
        return None
```

            'type': '夏长确认买',
            'buy_price': round(buy_price, 2),
            'stop_loss': round(stop_loss, 2),
            'target_price': round(target_price, 2),
            'add_position': round(add_position, 2),
            'final_position': 0.70
        }

class AutumnSellPointCalculator:
    """秋收期卖点精确计算"""
    
    def calculate_first_sell(self, stock_data):
        """
        秋收第一卖：高位放量滞涨
        """
        # 信号1：价格新高但成交量>前高量1.5倍
        if stock_data['当前价格'] > stock_data['前高'] and \
           stock_data['成交量'] > stock_data['前高量'] * 1.5:
            
            # 减仓30%
            sell_position = 0.30
            
            return {
                'type': '秋收第一卖',
                'signal': '高位放量滞涨',
                'sell_position': sell_position,
                'remaining_position': 0.40
            }
        
        return None
    
    def calculate_second_sell(self, stock_data):
        """
        秋收第二卖：双顶跌破颈线
        """
        # 颈线 = 两个顶之间的低点
        neckline = stock_data['双底颈线']
        
        # 跌破颈线确认（收盘低于颈线3%）
        if stock_data['收盘价'] < neckline * 0.97:
            
            # 清仓剩余仓位
            return {
                'type': '秋收第二卖',
                'signal': '双顶跌破颈线',
                'sell_position': '全部清仓',
                'stop_price': round(neckline * 0.97, 2)
            }
        
        return None
```
